SELECT *
FROM Водители
WHERE Стаж NOT IN (3, 5, 6, 10) AND место_прописки LIKE 'еее'

SELECT Водители.номер_паспорта
FROM Водители
LEFT JOIN Диспетчерская
ON Водители.код_водителя=Диспетчерская.код_водителя
WHERE Диспетчерская.время_отбытия IS NULL


SELECT Автомобили.регистрационный_номер_авто, Диспетчерская.время_прибытия, Диспетчерская.время_отбытия, Маршрут.пункт_назначения
FROM Автомобили
JOIN Диспетчерская
ON Автомобили.код_авто=Диспетчерская.код_авто
JOIN Маршрут
ON Диспетчерская.путёвка=Маршрут.код_маршрута
WHERE Автомобили.регистрационный_номер_авто = 2015
